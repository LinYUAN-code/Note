3.运行Hello World 程序 ⚡ 注意吧项目 配置为Debug x64

```c++
#include <stdio.h>
#include <stdlib.h>
#include <uv.h>

int main() {
    uv_loop_t* loop = (uv_loop_t*) malloc(sizeof(uv_loop_t));
    uv_loop_init(loop);

    printf("Hello World From LRJ Using Libuv.\n");
    uv_run(loop, UV_RUN_DEFAULT);

    uv_loop_close(loop);
    free(loop);
    return 0;
}
```

4.你已经可以看到正确的结果了！
